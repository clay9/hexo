** elisp调用bash
   #+BEGIN_EXAMPLE
   (setq my-command
      (concat "IFS=: read -ra dirs <<<\"$PATH\"\n"
              "for dir in ${dirs[@]}; do\n"
              " echo got dir \"$dir\"\n"
              "done\n"))
   (shell-command (format "bash -c %s" (shell-quote-argument my-command)))
